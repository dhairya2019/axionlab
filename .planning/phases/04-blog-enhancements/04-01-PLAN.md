**Step 1: Add preProcess and postProcess rehype plugins to next.config.mjs**

Add `import { visit } from 'unist-util-visit'` at the top of next.config.mjs.

Define two inline rehype plugin functions BEFORE the `withMDX` call:

```javascript
const preProcess = () => (tree) => {
  visit(tree, (node) => {
    if (node?.type === 'element' && node?.tagName === 'pre') {
      const [codeEl] = node.children
      if (codeEl?.tagName !== 'code') return
      node.raw = codeEl.children?.[0]?.value ?? ''
    }
  })
}

const postProcess = () => (tree) => {
  visit(tree, 'element', (node) => {
    if (node?.type === 'element' && node?.tagName === 'pre') {
      node.properties['raw'] = node.raw
    }
  })
}
```

Update the `rehypePlugins` array to bracket `rehype-pretty-code`:
```javascript
rehypePlugins: [
  rehypeSlug,
  preProcess,          // MUST be BEFORE rehype-pretty-code
  [rehypePrettyCode, { theme: 'github-dark', keepBackground: false }],
  postProcess,         // MUST be AFTER rehype-pretty-code
],
```

CRITICAL ORDERING: `preProcess` captures raw text BEFORE syntax highlighting tokenizes it into `<span>` elements. `postProcess` promotes the stored raw text to an HTML property AFTER highlighting is done. Reversing the order causes `raw` to be `undefined`.

```typescript
'use client'

import { useState } from 'react'
import { Copy, Check } from 'lucide-react'

export function CopyButton({ text }: { text: string }) {
  const [copied, setCopied] = useState(false)

  const handleCopy = async () => {
    try {
      await navigator.clipboard.writeText(text)
      setCopied(true)
      setTimeout(() => setCopied(false), 2000)
    } catch {
      // Clipboard API unavailable (non-HTTPS dev, iframe restriction, permission denied)
    }
  }

  return (
    <button
      onClick={handleCopy}
      aria-label={copied ? 'Copied to clipboard' : 'Copy code to clipboard'}
      className="absolute top-3 right-3 p-1.5 text-muted hover:text-white transition-colors opacity-0 group-hover:opacity-100 focus:opacity-100"
    >
      {copied
        ? <Check size={14} className="text-accent" />
        : <Copy size={14} />
      }
    </button>
  )
}
```

Key details:
- `'use client'` directive required — `navigator.clipboard` is browser-only
- `opacity-0 group-hover:opacity-100` — button is invisible until hovering the code block's `group` container
- `focus:opacity-100` — keyboard-accessible (visible when focused)
- `text-muted` / `text-accent` / `text-white` — uses AXIONLAB design tokens
- `Copy` and `Check` from `lucide-react` (already installed)
- Silent catch — no fallback needed for non-HTTPS contexts

**Step 1: Create components/blog/callouts.tsx**

```typescript
import type { ReactNode } from 'react'

interface CalloutProps {
  children: ReactNode
  label: string
  borderColor: string
  labelColor: string
}

function Callout({ children, label, borderColor, labelColor }: CalloutProps) {
  return (
    <div className={`not-prose border-l-2 ${borderColor} bg-surface px-5 py-4 my-6`}>
      <p className={`text-[9px] font-black uppercase tracking-[0.4em] ${labelColor} mb-3`}>
        {label}
      </p>
      <div className="text-sm text-white/80 leading-relaxed">
        {children}
      </div>
    </div>
  )
}

export function Info({ children }: { children: ReactNode }) {
  return (
    <Callout label="Info" borderColor="border-white/30" labelColor="text-white/50">
      {children}
    </Callout>
  )
}

export function Warning({ children }: { children: ReactNode }) {
  return (
    <Callout label="Warning" borderColor="border-accent" labelColor="text-accent">
      {children}
    </Callout>
  )
}

export function Tip({ children }: { children: ReactNode }) {
  return (
    <Callout label="Tip" borderColor="border-white/15" labelColor="text-muted">
      {children}
    </Callout>
  )
}
```

Key AXIONLAB design constraints applied:
- `not-prose` on outer div — isolates from @tailwindcss/typography `prose` margins. If `not-prose` does not work in Tailwind v4, fall back to adding `[&_p]:m-0` on the inner div.
- No `rounded-*` classes — 0px border-radius is enforced site-wide
- No `shadow-*` classes — no box-shadow by design
- `bg-surface` — uses #111111 token from @theme
- `border-accent` — uses #ff1f3d for Warning variant
- `text-muted` — uses #666666 for Tip label
- `text-[9px]` with `tracking-[0.4em]` uppercase — matches the micro-label pattern used throughout AXIONLAB (same as tag labels on blog posts)
- `Callout` is a private helper, not exported — only `Info`, `Warning`, `Tip` are public
