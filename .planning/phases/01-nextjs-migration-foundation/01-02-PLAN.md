```typescript
import { Inter, Inter_Tight } from 'next/font/google'
import './globals.css'
import type { Metadata } from 'next'
import Nav from '@/components/Nav'
import Footer from '@/components/Footer'
import { Chatbot } from '@/components/Chatbot'

const inter = Inter({
  subsets: ['latin'],
  variable: '--font-inter',
  display: 'swap',
})

const interTight = Inter_Tight({
  subsets: ['latin'],
  weight: ['700', '800', '900'],
  variable: '--font-inter-tight',
  display: 'swap',
})

export const metadata: Metadata = {
  title: 'AXIONLAB | Engineering for the obsessed.',
  description: 'Independent systems engineering lab designing commerce infrastructure and high-performance applications.',
}

export default function RootLayout({ children }: { children: React.ReactNode }) {
  return (
    <html lang="en" className={`${inter.variable} ${interTight.variable}`}>
      <body className="bg-background text-white selection:bg-accent selection:text-white overflow-x-hidden">
        {/* Hash URL backward compatibility — runs before React hydration */}
        <script dangerouslySetInnerHTML={{
          __html: `
            (function() {
              var hash = window.location.hash;
              if (hash.startsWith('#/') && hash.length > 2) {
                var path = hash.slice(1);
                window.location.replace(path + window.location.search);
              }
            })();
          `
        }} />
        <Nav />
        <main className="min-h-screen">
          {children}
        </main>
        <Footer />
        <Chatbot />
      </body>
    </html>
  )
}
```

Key details:
1. **next/font CSS variables:** `variable: '--font-inter'` and `variable: '--font-inter-tight'` — these MUST match the var() references in globals.css @theme block (`--font-sans: var(--font-inter)`). The chain: next/font declaration -> CSS variable on <html> -> @theme reference -> Tailwind font-sans class.

2. **Inter_Tight weights:** Only `['700', '800', '900']` — matches the current CDN import which loads wght@700;800;900 only.

3. **Hash redirect script:** Uses `dangerouslySetInnerHTML` on a `<script>` tag — this is correct for App Router Server Components. It runs synchronously before hydration. Do NOT use `next/script` with `strategy="afterInteractive"` — that would cause a flash of wrong content.

4. **Chatbot import:** Named import `{ Chatbot }` because Chatbot.tsx uses `export const Chatbot: React.FC` (named export per codebase conventions).

5. **Body classes:** `bg-background` resolves to `#080808` via the @theme block. `selection:bg-accent` for red text selection.

Do NOT add any Google Fonts CDN `<link>` tags — next/font replaces those entirely.
Do NOT add `suppressHydrationWarning` unless hydration errors appear during testing.
